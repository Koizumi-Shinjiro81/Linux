#### 스크립트 설명
cd files/ssh_passwordless
sh ssh_passwordless.sh

export PREFIX=`cat /var/tmp/prefix`
HOSTS=${PREFIX}/src/tmp/hostlist
HOSTNAME=`hostname`
	↓/usr/local/src/tmp/hostlist 경로에 해당파일이 없을경우
	mkdir -p ${PREFIX}/src/tmp
        vi $HOSTS
OS_VER=8
rpm -ivh sshpass-1.09-4.el8.x86_64.rpm
sshpass -p ${PW} ssh $i "cat /dev/null > /etc/ssh/ssh_known_hosts2"
IPAD=`cat /etc/hosts | grep $i | awk '{print $1}'`
/usr/bin/ssh-keyscan -t ecdsa $i,$IPAD >> /etc/ssh/ssh_known_hosts2
	##조건문 설명
	↓이 파일이 존재할 경우 => /root/.ssh/id_ecdsa
	sshpass -p ${PW} ssh $i "rm -rf /root/.ssh"
sshpass -p ${PW} ssh $i "/usr/bin/ssh-keygen -f /root/.ssh/id_ecdsa -t ecdsa -N ''"
sshpass -p ${PW} ssh $i cat /root/.ssh/id_ecdsa.pub >> /root/.ssh/authorized_keys
cp $HOSTS /etc/ssh/shosts.equiv
cp $HOSTS /root/.shosts
cp $HOSTS /etc/skel/.shosts
cp /etc/ssh/ssh_config /etc/ssh/ssh_config_`date +"%F_%H:%M:%S"`
sed -i 's/#   HostbasedAuthentication no/HostbasedAuthentication yes/g' /etc/ssh/ssh_config
	##조건문 설명
	↓/etc/ssh/ssh_config 파일에서 EnableSSHKeysign yes 이게 있는지 확인, 없으면
	echo "EnableSSHKeysign yes" >> /etc/ssh/ssh_config
cp /etc/ssh/sshd_config /etc/ssh/sshd_config_`date +"%F_%H:%M:%S"`
sed -i 's/#HostbasedAuthentication no/HostbasedAuthentication yes/g' /etc/ssh/sshd_config
	##조건문 설명
	↓/etc/ssh/sshd_config 파일에서 HostbasedUsesNameFromPacketOnly 해당내용 없으면
	echo "HostbasedUsesNameFromPacketOnly yes" >> /etc/ssh/sshd_config
service sshd restart
