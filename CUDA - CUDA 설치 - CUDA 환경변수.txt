# CUDNN URL
https://developer.nvidia.com/cudnn

### cuda.sh , cuda.csh
cat << EOF > /etc/profile.d/cuda.sh
#!/bin/bash
export LD_LIBRARY_PATH=/usr/local/cuda/lib64:/usr/local/cuda/extras/CUPTI/lib64:\$LD_LIBRARY_PATH
export CUDA_HOME=/usr/local/cuda
export NVHPC_CUDA_HOME=/usr/local/cuda
export PATH=/usr/local/cuda/bin:\$PATH 
EOF

cat << EOF > /etc/profile.d/cuda.csh
#!/bin/csh
setenv LD_LIBRARY_PATH /usr/local/cuda/lib64:/usr/local/cuda/extras/CUPTI/lib64
setenv CUDA_HOME /usr/local/cuda
setenv NVHPC_CUDA_HOME /usr/local/cuda
setenv PATH /usr/local/cuda/bin:\$PATH
EOF
